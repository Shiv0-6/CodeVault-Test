# Write your MySQL query statement below# Write your MySQL query statement below
SELECTSELECT
    e.name AS Employee    e.name AS Employee
FROMFROM
    Employee e    Employee e
JOINJOIN
    Employee m ON e.managerId = m.id    Employee m ON e.managerId = m.id
WHEREWHERE
    e.salary > m.salary;    e.salary > m.salary;
